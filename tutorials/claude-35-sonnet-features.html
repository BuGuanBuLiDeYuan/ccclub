        const primaryNode = this.nodes.getNode(key);
        const replicaNodes = this.nodes.getReplicaNodes(key, this.replicationFactor);
        
        // å¹¶è¡Œå†™å…¥ä¸»èŠ‚ç‚¹å’Œå‰¯æœ¬èŠ‚ç‚¹
